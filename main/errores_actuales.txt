 struct Maquina_de_estados_protesis
 {
     enum Estado_Protesis estado_actual; ///< Estado actual de la prótesis.
     enum Fase_Estado fase_actual;       ///< Fase actual dentro del estado.
 };

 /**
  * @brief Inicializa la máquina de estados en @ref ESTADO_NORMAL y @ref FASE_1.
  * @param maquina Puntero a la estructura que se inicializará.
  */
 static inline void maquina_inicializar(struct Maquina_de_estados_protesis *maquina)
 {
     maquina->estado_actual = ESTADO_NORMAL;
     maquina->fase_actual = FASE_1;
 }

 /**
  * @brief Cambia el estado de la prótesis y reinicia la fase a @ref FASE_1.
  * @param maquina Puntero a la máquina de estados.
  * @param nuevo_estado Nuevo estado que tomará la prótesis.
  */
 static inline void maquina_cambiarEstado(struct Maquina_de_estados_protesis *maquina, enum Estado_Protesis nuevo_estado)
 {
     maquina->estado_actual = nuevo_estado;
     maquina->fase_actual = FASE_1;
 }

 /**
  * @brief Cambia la fase actual de la prótesis.
  * @param maquina Puntero a la máquina de estados.
  * @param nueva_fase Nueva fase que se establecerá.
  */
 static inline void maquina_cambiarFase(struct Maquina_de_estados_protesis *maquina, enum Fase_Estado nueva_fase)
 {
     maquina->fase_actual = nueva_fase;
 }
